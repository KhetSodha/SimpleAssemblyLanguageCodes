.model small
.stack 100h
.data
	PROMPT DB 'Please enter 15 characters randomly:',0DH,0AH,'$'

	VOWELS DB ?
	LETTERS DB ?
	STV DB 'NUMBER OF VOWELS:','$'
	STB DB 'NUMBER OF NON-VOWELS:$'

.code

main proc
MOV AX,@DATA
MOV DS,AX

MOV AH,09
LEA DX,PROMPT
INT 21H

MOV AH,01

START_:
	INT 21H
	
	CMP AL,0DH
	JE PRINT	

	CMP AL,'a'
	JE VOWEL

	CMP AL,'e'
	JE VOWEL

	CMP AL,'i'
	JE VOWEL
	
	CMP AL,'o'
	JE VOWEL
	
	CMP AL,'u'
	JE VOWEL
	
	JMP NONV

VOWEL:
	INC VOWELS
	JMP START_

NONV:
	INC LETTERS
	JMP START_

	
PRINT:
	MOV AH,02H
	MOV DL,0DH
	INT 21H

	MOV AH,02H
	MOV DL,0AH
	INT 21H

	MOV AH,09
	LEA DX, STV
	INT 21H
	
	ADD VOWELS,30H
	
	MOV AH,02H
	MOV DL,VOWELS
	INT 21H
	
	MOV AH,02H
	MOV DL,0DH
	INT 21H

	MOV AH,02H
	MOV DL,0AH
	INT 21H
	
	MOV AH,09
	LEA DX, STB
	INT 21H
	
	ADD LETTERS,30H
	MOV AH,02H
	MOV DL,LETTERS
	INT 21H

MOV AH,4CH
INT 21H

MAIN ENDP
END MAIN
